//7 - Calcule o IMC. De acordo com (b), o IMC=peso em kilos dividido pela altura
//(em metros) ao quadrado.
//(a) http://como-emagrecer.com/calculo-de-imc.html
//(b) http://como-emagrecer.com/calculo-imc.html
//Leia a altura (em metros) e o peso em kilos de uma pessoa, calcule o IMC e
//indique a situação de acordo com a tabela em (a). (Abaixo do peso, peso ideal, sobrepeso, obesidade I, II, III)

#include <stdio.h>
#include<stdlib.h>

int main(int argc, char** argv)
{
    float peso, altura, imc;

    printf("Digite o seu peso:");
    scanf("%f",&peso);

    printf("Digite a sua altura:");
    scanf("%f",&altura);

    imc=peso/(altura*altura);

    printf("O seu imc é: %.f", imc);

    if(imc<18.5) {
        printf("Você está abaixo do peso.");
    }
    if(imc>=18.5 && imc<=24.9) {
        printf("\nParabéns, você está em seu peso normal.\n");
    }
    if(imc>=25 && imc<=29.9) {
        printf("\nVocê está acima de seu peso (sobrepeso).\n");
    }
    if(imc>=30 && imc<=34.9) {
        printf("\nObesidade grau I.\n");
    }
    if(imc>=35.0 && imc<=39.9) {
        printf("\nObesidade grau II.\n");
    }
    if(imc>=40) {
        printf("\nObesidade grau III.\n");
    }

    return 0;
}
